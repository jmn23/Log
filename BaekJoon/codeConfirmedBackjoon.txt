	scanf("%d %d", &n, &m);

	for(i=0;i<n;i++)
	{
		bucket[i] = i+1;
	}
	for(i=0;i<m;i++)
	{
		scanf("%d %d", &a, &b);
		for(j=0;j<=(b-a)/2;j++)
		{
			temp = bucket[a + j - 1];
			bucket[a + j - 1] = bucket[b - j - 1];
			bucket[b - j - 1] = temp;
		}
	
	}
	for (i = 0;i < n;i++)
	{
		printf("%d ", bucket[i]);
	}
	

	return 0;
}

#include <stdio.h>

int main() 
{
	double exam[1000];
	double avg;
	
	int i,n;
	double m = 0;
	double sum = 0;
	scanf("%d", &n);
	for(i=0;i<n;i++)
	{		
		scanf("%lf", &exam[i]);
		if(exam[i]>m)
		{
			m = exam[i];
		}
	}
	for (i = 0;i < n;i++)
	{
		sum += exam[i] / m * 100;
	}
	avg = sum / n;
	printf("%lf", avg);

	return 0;
}

		}
